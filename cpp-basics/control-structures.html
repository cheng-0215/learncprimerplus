控制结构 ​

控制结构是编程语言中用于控制程序执行流程的语句。C++提供了三种基本的控制结构：顺序结构、选择结构（条件语句）和循环结构。本章将详细介绍C++中的各种控制结构。

选择结构（条件语句） ​

选择结构允许程序根据条件执行不同的代码块。C++提供了以下几种选择结构：

if 语句 ​

if语句是最基本的条件语句，用于在条件为真时执行特定代码块：

cpp
if (条件) {
    // 当条件为真时执行的代码
}

例如：

cpp
int age = 20;

if (age >= 18) {
    cout << "你已经成年了" << endl;
}

if-else 语句 ​

if-else语句允许在条件为真时执行一个代码块，在条件为假时执行另一个代码块：

cpp
if (条件) {
    // 当条件为真时执行的代码
} else {
    // 当条件为假时执行的代码
}

例如：

cpp
int age = 15;

if (age >= 18) {
    cout << "你已经成年了" << endl;
} else {
    cout << "你还未成年" << endl;
}

if-else if-else 语句 ​

if-else if-else语句用于测试多个条件：

cpp
if (条件1) {
    // 当条件1为真时执行的代码
} else if (条件2) {
    // 当条件1为假且条件2为真时执行的代码
} else if (条件3) {
    // 当条件1和条件2都为假且条件3为真时执行的代码
} else {
    // 当所有条件都为假时执行的代码
}

例如：

cpp
int score = 85;

if (score >= 90) {
    cout << "优秀" << endl;
} else if (score >= 80) {
    cout << "良好" << endl;
} else if (score >= 60) {
    cout << "及格" << endl;
} else {
    cout << "不及格" << endl;
}

嵌套的 if 语句 ​

if语句可以嵌套在另一个if或else语句中：

cpp
if (条件1) {
    if (条件2) {
        // 当条件1和条件2都为真时执行的代码
    } else {
        // 当条件1为真但条件2为假时执行的代码
    }
} else {
    // 当条件1为假时执行的代码
}

例如：

cpp
int age = 25;
bool hasID = true;

if (age >= 18) {
    if (hasID) {
        cout << "可以进入" << endl;
    } else {
        cout << "需要出示身份证" << endl;
    }
} else {
    cout << "未成年不能进入" << endl;
}

switch 语句 ​

switch语句用于根据表达式的值选择执行不同的代码块：

cpp
switch (表达式) {
    case 值1:
        // 当表达式等于值1时执行的代码
        break;
    case 值2:
        // 当表达式等于值2时执行的代码
        break;
    ...
    default:
        // 当表达式不等于任何case值时执行的代码
}

例如：

cpp
int day = 3;
switch (day) {
    case 1:
        cout << "星期一" << endl;
        break;
    case 2:
        cout << "星期二" << endl;
        break;
    case 3:
        cout << "星期三" << endl;
        break;
    case 4:
        cout << "星期四" << endl;
        break;
    case 5:
        cout << "星期五" << endl;
        break;
    case 6:
        cout << "星期六" << endl;
        break;
    case 7:
        cout << "星期日" << endl;
        break;
    default:
        cout << "无效的日期" << endl;
}

注意

  • switch语句中的表达式必须是整型或枚举类型，或者能够隐式转换为整型的类型
  • 每个case后面必须跟一个常量表达式
  • 如果没有break语句，执行会继续到下一个case（这称为"fall through"）
  • default分支是可选的

条件运算符（三元运算符） ​

条件运算符?:提供了一种简洁的方式来编写简单的条件表达式：

cpp
条件 ? 表达式1 : 表达式2

如果条件为真，整个表达式的值为表达式1的值；否则为表达式2的值。

cpp
int a = 10, b = 20;
int max = (a > b) ? a : b;  // max = 20

// 等价于
int max;
if (a > b) {
    max = a;
} else {
    max = b;
}

循环结构 ​

循环结构允许重复执行一段代码，直到满足特定条件。C++提供了以下几种循环结构：

while 循环 ​

while循环在条件为真时重复执行代码块：

cpp
while (条件) {
    // 循环体
}

例如：

cpp
int i = 1;
while (i <= 5) {
    cout << i << " ";
    i++;
}
// 输出：1 2 3 4 5

do-while 循环 ​

do-while循环与while循环类似，但它保证循环体至少执行一次：

cpp
do {
    // 循环体
} while (条件);

例如：

cpp
int i = 1;
do {
    cout << i << " ";
    i++;
} while (i <= 5);
// 输出：1 2 3 4 5

do-while循环和while循环的主要区别在于，do-while循环会先执行循环体，然后再检查条件：

cpp
int i = 10;

// while循环
while (i < 5) {
    cout << i << " ";
    i++;
}
// 不会输出任何内容，因为条件一开始就为假

// do-while循环
i = 10;
do {
    cout << i << " ";
    i++;
} while (i < 5);
// 输出：10
// 即使条件为假，循环体也会执行一次

for 循环 ​

for循环提供了一种更结构化的方式来编写循环：

cpp
for (初始化; 条件; 更新) {
    // 循环体
}

例如：

cpp
for (int i = 1; i <= 5; i++) {
    cout << i << " ";
}
// 输出：1 2 3 4 5

for循环的执行流程：

  1. 执行初始化语句（只执行一次）
  2. 检查条件，如果为假，退出循环
  3. 执行循环体
  4. 执行更新语句
  5. 返回步骤2

for循环的各个部分都是可选的：

cpp
// 省略初始化
int i = 1;
for (; i <= 5; i++) {
    cout << i << " ";
}

// 省略条件（无限循环，需要在循环体内部使用break退出）
for (int i = 1;; i++) {
    if (i > 5) break;
    cout << i << " ";
}

// 省略更新（在循环体内部更新）
for (int i = 1; i <= 5;) {
    cout << i << " ";
    i++;
}

// 省略所有部分（无限循环）
for (;;) {
    // 无限循环，需要使用break退出
}

基于范围的 for 循环（C++11） ​

C++11引入了基于范围的for循环，用于遍历容器或数组中的元素：

cpp
for (声明 : 表达式) {
    // 循环体
}

例如：

cpp
// 遍历数组
int arr[] = {1, 2, 3, 4, 5};
for (int x : arr) {
    cout << x << " ";
}
// 输出：1 2 3 4 5

// 遍历容器
vector<int> vec = {10, 20, 30, 40, 50};
for (int x : vec) {
    cout << x << " ";
}
// 输出：10 20 30 40 50

// 使用auto自动推导类型
for (auto x : vec) {
    cout << x << " ";
}

// 使用引用避免复制（如果需要修改元素）
for (auto& x : vec) {
    x *= 2;  // 将每个元素乘以2
}

跳转语句 ​

跳转语句用于改变程序的执行流程。

break 语句 ​

break语句用于提前退出循环或switch语句：

cpp
for (int i = 1; i <= 10; i++) {
    if (i == 6) {
        break;  // 当i等于6时退出循环
    }
    cout << i << " ";
}
// 输出：1 2 3 4 5

continue 语句 ​

continue语句用于跳过当前循环迭代的剩余部分，直接进入下一次迭代：

cpp
for (int i = 1; i <= 10; i++) {
    if (i % 2 == 0) {
        continue;  // 跳过偶数
    }
    cout << i << " ";
}
// 输出：1 3 5 7 9

goto 语句 ​

goto语句用于无条件跳转到程序中标记的位置：

cpp
int i = 1;
start:
if (i <= 5) {
    cout << i << " ";
    i++;
    goto start;
}
// 输出：1 2 3 4 5

注意

goto语句通常被认为是不好的编程实践，因为它会使代码难以理解和维护。在大多数情况下，可以使用其他控制结构（如循环和条件语句）代替goto。

return 语句 ​

return语句用于从函数返回值并结束函数的执行：

cpp
int max(int a, int b) {
    if (a > b) {
        return a;  // 返回a并结束函数
    } else {
        return b;  // 返回b并结束函数
    }
}

异常处理 ​

异常处理是一种处理程序运行时错误的机制。C++使用try、catch和throw关键字进行异常处理：

cpp
#include <iostream>
#include <stdexcept>  // 包含标准异常类
using namespace std;

int divide(int a, int b) {
    if (b == 0) {
        throw runtime_error("除数不能为零");  // 抛出异常
    }
    return a / b;
}

int main() {
    try {
        cout << divide(10, 2) << endl;  // 正常执行
        cout << divide(10, 0) << endl;  // 抛出异常
    } catch (const runtime_error& e) {
        cout << "捕获到异常: " << e.what() << endl;
    }
    
    cout << "程序继续执行" << endl;
    
    return 0;
}

异常处理的详细内容将在后续章节中介绍。
