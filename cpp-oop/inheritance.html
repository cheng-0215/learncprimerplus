继承 ​

继承是面向对象编程的三大特性之一（另外两个是封装和多态），它允许创建一个新类，从现有的类中获取属性和方法。本章将详细介绍C++中继承的概念和使用方法。

继承的基本概念 ​

在继承关系中，已有的类称为基类（或父类、超类），新创建的类称为派生类（或子类）。派生类继承了基类的特性，同时可以添加新的特性或修改继承的特性。

继承的语法 ​

cpp
class 派生类名 : 继承方式 基类名 {
    // 派生类的成员
};

其中，继承方式可以是：

  • public：公有继承
  • protected：保护继承
  • private：私有继承

如果不指定继承方式，默认为私有继承。

继承方式的影响 ​

继承方式决定了基类成员在派生类中的访问权限：

基类成员公有继承保护继承私有继承
publicpublicprotectedprivate
protectedprotectedprotectedprivate
private不可访问不可访问不可访问

公有继承 ​

公有继承是最常用的继承方式，它保持基类成员的访问权限不变（除了private成员，它们在派生类中不可访问）。

cpp
class Shape {
public:
    void setWidth(int w) {
        width = w;
    }
    
    void setHeight(int h) {
        height = h;
    }
    
protected:
    int width;
    int height;
};

class Rectangle : public Shape { // 公有继承
public:
    int getArea() {
        return width * height; // 可以访问基类的protected成员
    }
};

int main() {
    Rectangle rect;
    
    rect.setWidth(5);  // 可以访问基类的public成员
    rect.setHeight(7);
    
    cout << "Area: " << rect.getArea() << endl;
    
    return 0;
}

保护继承 ​

保护继承将基类的public成员变为派生类的protected成员，这意味着它们只能在派生类及其派生类中访问，而不能在类外部访问。

cpp
class Shape {
public:
    void setWidth(int w) {
        width = w;
    }
    
    void setHeight(int h) {
        height = h;
    }
    
protected:
    int width;
    int height;
};

class Rectangle : protected Shape { // 保护继承
public:
    int getArea() {
        return width * height;
    }
    
    // 提供公有方法来设置宽度和高度
    void setDimensions(int w, int h) {
        setWidth(w);  // 可以访问基类的方法
        setHeight(h);
    }
};

int main() {
    Rectangle rect;
    
    // rect.setWidth(5);  // 错误！基类的public成员在派生类中变为protected
    rect.setDimensions(5, 7); // 正确，通过派生类的公有方法
    
    cout << "Area: " << rect.getArea() << endl;
    
    return 0;
}

私有继承 ​

私有继承将基类的所有成员（除了private成员）变为派生类的private成员，这意味着它们只能在派生类中访问，而不能在派生类的派生类或类外部访问。

cpp
class Shape {
public:
    void setWidth(int w) {
        width = w;
    }
    
    void setHeight(int h) {
        height = h;
    }
    
protected:
    int width;
    int height;
};

class Rectangle : private Shape { // 私有继承
public:
    int getArea() {
        return width * height;
    }
    
    // 提供公有方法来设置宽度和高度
    void setDimensions(int w, int h) {
        setWidth(w);
        setHeight(h);
    }
};

class Square : public Rectangle { // 从Rectangle派生
public:
    void setSide(int side) {
        // setWidth(side);  // 错误！基类的方法在Rectangle中变为private
        setDimensions(side, side); // 正确，通过Rectangle的公有方法
    }
};

多重继承 ​

C++支持多重继承，即一个类可以从多个基类继承。

cpp
class Base1 {
public:
    void func1() {
        cout << "Base1::func1" << endl;
    }
};

class Base2 {
public:
    void func2() {
        cout << "Base2::func2" << endl;
    }
};

class Derived : public Base1, public Base2 { // 多重继承
public:
    void func3() {
        cout << "Derived::func3" << endl;
    }
};

int main() {
    Derived d;
    
    d.func1(); // 来自Base1
    d.func2(); // 来自Base2
    d.func3(); // 来自Derived
    
    return 0;
}

菱形继承问题 ​

多重继承可能导致菱形继承问题，即一个类通过多条路径继承自同一个基类，导致基类的成员在派生类中出现多次。

cpp
class Animal {
public:
    void eat() {
        cout << "Animal eating" << endl;
    }
};

class Mammal : public Animal {
public:
    void breathe() {
        cout << "Mammal breathing" << endl;
    }
};

class Bird : public Animal {
public:
    void fly() {
        cout << "Bird flying" << endl;
    }
};

class Bat : public Mammal, public Bird { // 菱形继承
public:
    void sleep() {
        cout << "Bat sleeping" << endl;
    }
};

int main() {
    Bat b;
    
    // b.eat(); // 错误！不明确的调用，Bat从Mammal和Bird两条路径继承了eat方法
    
    b.Mammal::eat(); // 正确，指定路径
    b.Bird::eat();    // 正确，指定路径
    
    return 0;
}

虚拟继承 ​

虚拟继承可以解决菱形继承问题，它确保基类在派生类中只有一个实例。

cpp
class Animal {
public:
    void eat() {
        cout << "Animal eating" << endl;
    }
};

class Mammal : virtual public Animal { // 虚拟继承
public:
    void breathe() {
        cout << "Mammal breathing" << endl;
    }
};

class Bird : virtual public Animal { // 虚拟继承
public:
    void fly() {
        cout << "Bird flying" << endl;
    }
};

class Bat : public Mammal, public Bird {
public:
    void sleep() {
        cout << "Bat sleeping" << endl;
    }
};

int main() {
    Bat b;
    
    b.eat(); // 正确，Animal在Bat中只有一个实例
    
    return 0;
}

构造函数和析构函数在继承中的调用顺序 ​

在继承关系中，构造函数和析构函数的调用顺序如下：

  • 构造函数：先调用基类的构造函数，再调用派生类的构造函数
  • 析构函数：先调用派生类的析构函数，再调用基类的析构函数
cpp
class Base {
public:
    Base() {
        cout << "Base constructor" << endl;
    }
    
    ~Base() {
        cout << "Base destructor" << endl;
    }
};

class Derived : public Base {
public:
    Derived() {
        cout << "Derived constructor" << endl;
    }
    
    ~Derived() {
        cout << "Derived destructor" << endl;
    }
};

int main() {
    Derived d; // 输出：Base constructor, Derived constructor
    
    return 0; // 输出：Derived destructor, Base destructor
}

派生类构造函数中调用基类构造函数 ​

派生类可以在初始化列表中调用基类的构造函数。

cpp
class Base {
private:
    int value;
    
public:
    Base(int v) : value(v) {
        cout << "Base constructor, value = " << value << endl;
    }
};

class Derived : public Base {
private:
    int data;
    
public:
    Derived(int v, int d) : Base(v), data(d) {
        cout << "Derived constructor, data = " << data << endl;
    }
};

int main() {
    Derived d(10, 20);
    
    return 0;
}

总结 ​

  • 继承允许一个类从另一个类获取属性和方法
  • C++支持三种继承方式：公有继承、保护继承和私有继承
  • 公有继承是最常用的继承方式，它保持基类成员的访问权限不变
  • C++支持多重继承，但可能导致菱形继承问题
  • 虚拟继承可以解决菱形继承问题
  • 在继承关系中，构造函数的调用顺序是从基类到派生类，析构函数的调用顺序是从派生类到基类
