C++基本语法参考 ​

本文档提供了C++语言的基本语法参考，方便您在编程过程中快速查阅。

基本程序结构 ​

cpp
#include <iostream>

int main() {
    // 程序代码
    std::cout << "Hello, World!" << std::endl;
    return 0;
}

注释 ​

cpp
// 单行注释

/* 
多行注释
可以跨越多行
*/

数据类型 ​

类型关键字描述
整型int整数
字符型char单个字符
布尔型booltrue 或 false
浮点型float单精度浮点数
双精度浮点型double双精度浮点数
无类型void表示无类型
宽字符型wchar_t宽字符

变量声明 ​

cpp
int a;           // 声明变量
int b = 10;      // 声明并初始化
const int c = 5; // 常量声明

运算符 ​

算术运算符 ​

cpp
a + b  // 加法
a - b  // 减法
a * b  // 乘法
a / b  // 除法
a % b  // 取模（余数）
a++    // 自增
a--    // 自减

关系运算符 ​

cpp
a == b // 等于
a != b // 不等于
a > b  // 大于
a < b  // 小于
a >= b // 大于等于
a <= b // 小于等于

逻辑运算符 ​

cpp
a && b // 逻辑与
a || b // 逻辑或
!a     // 逻辑非

控制结构 ​

条件语句 ​

cpp
// if 语句
if (条件) {
    // 代码
} else if (条件) {
    // 代码
} else {
    // 代码
}

// switch 语句
switch (表达式) {
    case 常量1:
        // 代码
        break;
    case 常量2:
        // 代码
        break;
    default:
        // 代码
}

循环语句 ​

cpp
// for 循环
for (初始化; 条件; 增量) {
    // 代码
}

// while 循环
while (条件) {
    // 代码
}

// do-while 循环
do {
    // 代码
} while (条件);

函数 ​

cpp
// 函数声明
返回类型 函数名(参数列表);

// 函数定义
返回类型 函数名(参数列表) {
    // 函数体
    return 值; // 如果有返回值
}

// 示例
int add(int a, int b) {
    return a + b;
}

数组 ​

cpp
// 声明数组
类型 数组名[大小];

// 初始化数组
int numbers[5] = {1, 2, 3, 4, 5};

// 多维数组
int matrix[3][3] = {
    {1, 2, 3},
    {4, 5, 6},
    {7, 8, 9}
};

指针 ​

cpp
// 声明指针
类型* 指针名;

// 获取变量地址
int a = 10;
int* p = &a;

// 通过指针访问值
int value = *p;

引用 ​

cpp
// 声明引用
类型& 引用名 = 变量名;

// 示例
int a = 10;
int& ref = a;
ref = 20; // 现在 a 也等于 20

类和对象 ​

cpp
// 类定义
class 类名 {
private:
    // 私有成员

protected:
    // 受保护成员

public:
    // 公有成员
    // 构造函数
    类名(参数列表);
    
    // 析构函数
    ~类名();
    
    // 成员函数
    返回类型 函数名(参数列表);
};

// 创建对象
类名 对象名;
类名 对象名(参数);

命名空间 ​

cpp
// 定义命名空间
namespace 命名空间名 {
    // 代码
}

// 使用命名空间
using namespace 命名空间名;

// 或者直接使用
命名空间名::成员;

异常处理 ​

cpp
try {
    // 可能抛出异常的代码
    throw 异常;
} catch (异常类型1 变量) {
    // 处理异常类型1
} catch (异常类型2 变量) {
    // 处理异常类型2
} catch (...) {
    // 处理任何其他异常
}

预处理器指令 ​

cpp
#include <头文件>  // 包含标准头文件
#include "头文件" // 包含自定义头文件

#define 宏名 值    // 定义宏
#define 宏名(参数) 表达式 // 带参数的宏

#ifdef 宏名        // 如果宏已定义
#ifndef 宏名       // 如果宏未定义
#endif            // 结束条件编译块
